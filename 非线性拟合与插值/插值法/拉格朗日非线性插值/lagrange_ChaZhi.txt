x0=[0 3 5 7 9 11 12 13 14 15];
y0=[0 1.2 1.7 2.0 2.1 2.0 1.8 1.2 1.0 1.6];
x=0:0.1:15;
y1=lagrange(x0,y0,x); %前面编写的拉格朗日插值函数
y2=interp1(x0,y0,x);
y3=interp1(x0,y0,x,'spline');
[x',y1',y2',y3']
subplot(2,2,1)
plot(x0,y0,'+',x,y1)
title('Lagrange')
subplot(2,2,2)
plot(x0,y0,'+',x,y2)
title('Piecewise linear')
subplot(2,2,3)
plot(x0,y0,'+',x,y3)
title('Spline')
